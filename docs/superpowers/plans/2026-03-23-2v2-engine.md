```typescript
/**
 * Resolve an N-player RPS round (elimination style).
 * Returns null if tie (all 3 choices present, or all same choice).
 * Returns { winners, losers } if exactly 2 distinct choices.
 */
export function resolveRPSMulti(
  choices: Record<string, RPSChoice>,
): { winners: string[]; losers: string[] } | null {
  const players = Object.keys(choices);
  if (players.length <= 1) {
    return { winners: players, losers: [] };
  }

  const distinctChoices = new Set(Object.values(choices));
  // All same or all 3 present = tie
  if (distinctChoices.size !== 2) return null;

  const [choiceA, choiceB] = [...distinctChoices];
  const result = compareRPS(choiceA, choiceB);
  // Should never be 0 since choices are distinct
  const winningChoice = result === 1 ? choiceA : choiceB;

  const winners: string[] = [];
  const losers: string[] = [];
  for (const [id, choice] of Object.entries(choices)) {
    if (choice === winningChoice) {
      winners.push(id);
    } else {
      losers.push(id);
    }
  }

  return { winners, losers };
}
```

```typescript
/**
 * Create a new game state.
 * For backward compatibility, supports both old 1v1 signature and new multi-player signature.
 */
export function createGameState(
  gameId: string,
  modeOrPlayer1: '1v1' | '2v2' | '3v3' | { id: string; name: string; heroId: string },
  playersOrPlayer2?: { id: string; name: string; heroId: string; teamIndex?: number }[] | { id: string; name: string; heroId: string },
): GameState {
  // Handle legacy 1v1 signature: createGameState(id, player1, player2)
  if (typeof modeOrPlayer1 === 'object') {
    const player1 = modeOrPlayer1;
    const player2 = playersOrPlayer2 as { id: string; name: string; heroId: string };
    return createGameStateMulti(gameId, '1v1', [
      { ...player1, teamIndex: 0 },
      { ...player2, teamIndex: 1 },
    ]);
  }

  // New signature: createGameState(id, mode, players[])
  const mode = modeOrPlayer1;
  const players = playersOrPlayer2 as { id: string; name: string; heroId: string; teamIndex: number }[];
  return createGameStateMulti(gameId, mode, players);
}

function createGameStateMulti(
  gameId: string,
  mode: '1v1' | '2v2' | '3v3',
  players: { id: string; name: string; heroId: string; teamIndex: number }[],
): GameState {
  // Validate unique heroes
  const heroIds = players.map(p => p.heroId);
  if (new Set(heroIds).size !== heroIds.length) {
    throw new Error('Duplicate hero IDs not allowed');
  }

  // Validate 3v3 has enough heroes
  if (mode === '3v3') {
    const { getHeroIds } = await import('../heroes/registry.js');
    // Static check: 3v3 needs 6 unique heroes
    if (players.length < 6) {
      throw new Error('3v3 requires 6 players');
    }
  }

  const team0Players = players
    .filter(p => p.teamIndex === 0)
    .map(p => createPlayerState(p.id, p.name, p.heroId, TEAM_0_START));
  const team1Players = players
    .filter(p => p.teamIndex === 1)
    .map(p => createPlayerState(p.id, p.name, p.heroId, TEAM_1_START));

  const positionsAtTurnStart: Record<string, number> = {};
  for (const p of players) {
    positionsAtTurnStart[p.id] = p.teamIndex === 0 ? TEAM_0_START : TEAM_1_START;
  }

  return {
    id: gameId,
    mode,
    phase: 'rps_submit',
    turn: 1,
    teams: [
      { teamIndex: 0, players: team0Players },
      { teamIndex: 1, players: team1Players },
    ],
    positionsAtTurnStart,
    pendingRPS: {},
    actionOrder: [],
    currentActionIndex: 0,
    awaitingMinionAction: false,
    history: [],
    winner: null,
    stunImmuneThisTurn: [],
  };
}
```

```typescript
import { resolveRPS1v1, resolveRPSMulti, randomRPSChoice } from './rps.js';

export function resolveRPSRound(state: GameState): RPSResult {
  const allPlayers = getAllAlivePlayers(state);
  const nonStunnedPlayers = allPlayers.filter(p => !isStunned(p.hero));

  // Auto-fill missing choices with random
  for (const player of nonStunnedPlayers) {
    if (state.pendingRPS[player.id] == null) {
      state.pendingRPS[player.id] = randomRPSChoice();
    }
  }

  // Only one non-stunned player: they win automatically
  if (nonStunnedPlayers.length <= 1) {
    const winners = nonStunnedPlayers.map(p => p.id);
    const losers = allPlayers.filter(p => !winners.includes(p.id)).map(p => p.id);

    const result: RPSResult = {
      choices: { ...state.pendingRPS } as Record<string, RPSChoice>,
      winners,
      losers,
      draw: false,
    };

    state.phase = 'action_phase';
    state.actionOrder = winners;
    state.currentActionIndex = 0;
    state.pendingRPS = {};
    return result;
  }

  // Build choices map for non-stunned players
  const choices: Record<string, RPSChoice> = {};
  for (const p of nonStunnedPlayers) {
    choices[p.id] = state.pendingRPS[p.id]!;
  }

  const multiResult = resolveRPSMulti(choices);

  const allChoices = { ...state.pendingRPS } as Record<string, RPSChoice>;
  state.pendingRPS = {};

  if (!multiResult) {
    // Tie
    return { choices: allChoices, winners: [], losers: [], draw: true };
  }

  // Winners need ordering via additional RPS rounds.
  // For now, use the order they appear in multiResult.winners.
  // In a real game, the server would collect additional RPS submissions.
  // For the engine, we store the winners as actionOrder directly.
  const stunnedIds = allPlayers.filter(p => isStunned(p.hero)).map(p => p.id);

  state.phase = 'action_phase';
  state.actionOrder = multiResult.winners;
  state.currentActionIndex = 0;

  return {
    choices: allChoices,
    winners: multiResult.winners,
    losers: [...multiResult.losers, ...stunnedIds],
    draw: false,
  };
}
```
