=== Punch-only duel: 3-punch stun triggers
# Both at distance 0, trading punches.
# 3rd punch triggers stun — applied after damage, stays on.
heroes: shan vs shan
pos: 5 5

turn 1: p1 wins
  p1 punch p2
  > p2: hp=90 stunned=false

turn 2: p1 wins
  p1 punch p2
  > p2: hp=80 stunned=false

turn 3: p1 wins
  p1 punch p2
  > p2: hp=70 stunned=true

turn 4: p1 wins
  # startTurn ticks stun off. p1 acts.
  p1 punch p2
  > p2: hp=60 stunned=false
