=== Wind Walk approach: stealth + double speed movement
# Start at distance 3. Jin uses Wind Walk, moves at double speed (2/turn).
# Wind Walk lasts 3 rounds, then normal speed resumes.
heroes: jin vs shan
pos: 2 5

turn 1: p1 wins
  p1 skill wind_walk
  > p1: invisible=true pos=2

turn 2: p1 wins
  # Invisible speed 2: pos 2→4
  p1 move_forward
  > p1: pos=4 invisible=true

turn 3: p1 wins
  # Invisible speed 2: pos 4→6 (distance 1 from enemy at 5, legal)
  p1 move_forward
  > p1: pos=6 invisible=true

turn 4: p1 wins
  # Wind Walk ended (3 rounds). Normal speed 1.
  # From pos 6 toward enemy at 5: pos 5
  p1 move_forward
  > p1: pos=5 invisible=false
